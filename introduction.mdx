    Quick start
  </h2>
  <p className="text-base dark:text-gray-400 text-gray-600 mb-8">
    Get your Pet Adoption API up and running in minutes
  </p>

  <Steps>
    <Step title="Clone and install dependencies">
      Clone the repository and install required packages:
      
      ```bash
      git clone https://github.com/justinamutigliengo/proyecto_back.git
      cd proyecto_back
      npm install
      ```
    </Step>

    <Step title="Configure environment variables">
      Create a `.env` file in the root directory with your MongoDB connection string and JWT secret:
      
      ```bash
      MONGO_URL=mongodb://localhost:27017/adoptme
      PORT=8080
      JWT_SECRET=your-secret-key-here
      ```
    </Step>

    <Step title="Start the server">
      Run the development server:
      
      ```bash
      npm run dev
      ```
      
      The API will be available at `http://localhost:8080`
    </Step>

    <Step title="Make your first API call">
      Register a new user to get started:
      
      ```bash cURL
      curl -X POST http://localhost:8080/api/sessions/register \
        -H "Content-Type: application/json" \
        -d '{
          "first_name": "Jane",
          "last_name": "Doe",
          "email": "jane@example.com",
          "password": "securepass123",
          "role": "user"
        }'
      ```
      
      <Accordion title="Example response">
        ```json
        {
          "status": "success",
          "payload": {
            "_id": "507f1f77bcf86cd799439011",
            "first_name": "Jane",
            "last_name": "Doe",
            "email": "jane@example.com",
            "role": "user",
            "pets": []
          }
        }
        ```
      </Accordion>
    </Step>
  </Steps>
</div>
